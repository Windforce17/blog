    return (x << n) | (x >> (32-n))

    res = res + a + func(b, c, d)

    res = res + ac
    res = res & 0xffffffffL
    res = _rotateLeft(res, s)
    res = res & 0xffffffffL
    res = res + b

    def padding(self, n, sz=None):
        if sz is None:
            sz = n
        pre = 64-8
        sz = struct.pack("Q",sz*8)
        pad = '\x80'
        n += 1
        if n % 64 <= pre:
            pad += '\x00' * (pre - n % 64)
            pad += sz
        else:
            pad += '\x00' * (pre + 64 - n % 64)
            pad += sz
        return pad
